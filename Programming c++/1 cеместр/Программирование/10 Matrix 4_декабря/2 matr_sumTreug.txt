// Дана матрица a[n][n]. Вычислить сумму элементов
// нижнего треугольника матрицы и произведение элементов 
// главной диагонали матрицы.
#include <iostream>
using namespace std;
const int DIM = 20;
int main(){
     int a[DIM][DIM], n;
     cout<<" Enter n: ";		cin >> n;
    srand(n);
    for(int i = 0; i < n; i++)
      for(int j = 0; j < n; j++)
          a[i][j] = rand() % 20 - 8;

/*  cout<<"Vvedi matrix "<<endl;
    for(int i = 0; i < n; i++)
       for(int j = 0; j < n; j++)
           cin>>a[i][j];
*/  
     cout<<"\n\tMatrix = \n";
     for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
           cout.width(4);
          cout<<a[i][j];
        }
       cout << endl;
    }
   int sum = 0;
   for(int i = 1; i < n; i++)
       for(int j = 0; j < i ; j++)
           sum = sum + a[i][j];			// сумма элементов нижнего
      cout<<"sum = "<<sum<<endl;		// треугольника матрицы
    long mult = 1;
    for(int i = 0; i < n; i++)			// произведение элементов
        mult = mult * a[i][i];			// главной диагонали матрицы
    cout<<"mult = "<<mult<<endl;
    system("pause");
}
