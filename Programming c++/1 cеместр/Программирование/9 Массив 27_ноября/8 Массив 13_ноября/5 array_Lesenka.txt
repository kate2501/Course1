// Вывести массив лесенкой - слева направо и справа налево.
#include <iostream>
using namespace std;
const int DIM = 40; 
int main(){
int mas[DIM];    int n;
   cout<<" Enter size massiva n: ";   cin>>n;
   srand(n);

  for(int i = 0; i < n; i++)		// ввод массива

//      cin>>mas[i];
   cout<<"\nIsxodni massiv\n";		// вывод массива
   for(int i = 0; i < n; i++)
        cout<<mas[i]<< ' ';
   cout<<endl;
    cout<<"\n\tMassiv_Lesenka_1\n";	// вывод массива лесенкой слева направо 
    for(int i = 0; i < n; i++){

           cout<<' ';
       cout<<mas[i]<<endl;
    }
     cout<<endl;
    cout<<"\n\tMassiv_Lesenka_2\n";	// вывод массива лесенкой справа  налево 
    int k = n;

        for(int j = 0; j < k; j++)
