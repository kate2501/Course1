void funObmen(int mas[], int n){
   cout<<"\n ***funObmen***"<<endl;
   int  max = mas[0], min = mas[0];		// поиск максимального и
   int  imax = 0, imin = 0;			// минимального элементов массива
   for(int i = 0; i < n ; i++){			// и их индексов
       if(mas[i] < min){min = mas[i]; imin = i;}
       if(mas[i] > max){max = mas[i]; imax = i;}
    }
   cout<<" max = "<<max<<"  imax = "<<imax<<endl;		// для контроля
   cout<<" min = "<<min<<" imin = "<<imin<<endl;
   mas[imin] = max; mas[imax] = min;
 }
