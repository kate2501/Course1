void  funMaxMin(int mas[], int n, int& imax, int& imin){
     cout<<"\n ***funMax***"<<endl;
     int  max = mas[0], min = mas[0]; 
     imax = 0; imin = 0;				// поиск максимального и минимального 
     for(int i = 0; i < n ; i++){			// элементов  и их индексов
         if(mas[i] < min){min = mas[i]; imin = i;}
         if(mas[i] > max){max = mas[i]; imax = i;}
     }
 }
void funObmen(int mas[], int n, int imax, int imin){

    int rab = mas[imax];
    mas[imax] = mas[imin];
    mas[imin] = rab;  
 }
