void EnterArray(int *, int);
void ShowArray(int *, int);

int* FindMax(int*, int);
int* FindMin(int*, int);
int SumEl(int);
int DeleteEl(int*, int);

int main()
{
	int n;
	cout << "Enter n" << endl;
	cin >> n;
	int *myarray = new int[n];
	EnterArray(myarray,n);
	ShowArray(myarray, n);
	cout <<"\nmin: "<< *FindMin(myarray, n)<<endl;
	cout <<"max: "<< *FindMax(myarray, n)<<endl;
	int k=DeleteEl(myarray, n);
	ShowArray(myarray, k);
	delete[] myarray;
}

void EnterArray(int* myarray,int n)
{
	for (int i = 0; i < n; i++)
	{
		cout << "Enter the element" << endl;
		cin >> *(myarray + i);
	}
}
 
void ShowArray(int *arr, int n)
{
	cout << "\nThat is your array" << endl;
	for (int i = 0; i < n; i++)
	{
		cout<< *(arr + i)<<" ";
	}
	cout << endl;
}

bool CheckEvenQ(int n)
{
	if(n%2==0) return true;
	else return false;
}
int* FindMax(int *arr, int n)
{
	int *p = arr;
	for (int i = 0; i < n; i++)
	{
		if (*(arr + i) >= *p)
		{
			p = arr + i;
		}
	}
	return p;
}
int* FindMin(int *arr, int n)
{
	int *p = arr;
	for (int i = 0; i < n; i++)
	{
		if (*(arr + i) < *p)
		{
			p = arr + i;
		}
	}
	return p;
}
int SumEl(int n)
{
	int sum = 0;
	while (n != 0)
	{
		sum += n % 10;
		n /= 10;
	}
	return sum;
}
int DeleteEl(int *arr, int n)
{
	int *min = FindMin(arr, n);
	int *max = FindMax(arr, n);
	for (int*i = min+1; i <arr+n; i++) 
	{
		if (i < max) 
		{
			if (CheckEvenQ(SumEl(*i)))
			{
				for (int*j = i; j < arr + n; j++)
				{
					*j = *(j + 1);
				}
				n--;
				max--;
			}
		}
		else break;
	}
	return n;
}
