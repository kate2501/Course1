// Пример 4. Создание и удаление двумерного динамического массива.
#include <iostream>
using namespace std;
int main() {
	int n, m;
	cout << "Vvedi n, m:  ";
	cin >> n >> m;				// n – число строк, m – число столбцов
	int** p;				// p – указатель на указатель
	p = new int* [n]; 			// массив указателей на строки
	for (int i = 0; i < n; i++)		// выделение памяти для каждой строки
	   p[i] = new int[m];
	p[0][0] = 55;				// присвоили p[0][0] значение 55
	cout << " p = " << p << " *p = " << *p << '\n';
	cout << " **p = " << **p << '\n';
	cout << " p[0] = " << p[0] << '\n';
	cout << " &p[0][0] = " << &p[0][0] << '\n';
	cout << " p[1] = " << p[1] << '\n';
	cout << " &p[1][0] = " << &p[1][0] << '\n';		// (p[1] +0) или ((p +1)+0)
	for (int i = 0; i < n; i++)		// освобождение памяти
	   delete[] p[i];			// удаление строк
	delete[] p;				// удаление массива указателей
}
