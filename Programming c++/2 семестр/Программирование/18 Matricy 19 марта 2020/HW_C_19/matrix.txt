//В матрице А(N,N) найти первую строку, 
//не содержащую отрицательных элементов, и умножить её как вектор на матрицу А.
//Belskaya
#include "pch.h"
#include <iostream>
#include<iostream>
using namespace std;

int search(int **, int);

int main() 
{
	int n,k;
	cout << "Enter n:"; cin >> n;
	srand(n);
	int**a = new int*[n];
	int *p = new int[n];
	
	for (int i = 0; i < n; i++) 
	{
		a[i] = new int[n];            		
		for (int j = 0; j < n; j++)         		
			*(a[i] + j) = rand() % 25 - 2;
	}

	for (int i = 0; i < n; i++) 
	{
		for (int j = 0; j < n; j++) {
			cout.width(5);
			cout << a[i][j];
		}
		cout << endl;
	}

	k = search(a, n);
	for (int i = 0; i < n; i++)
	{
		p[i] = 0;
		for (int j = 0; j < n; j++) 
		{
			p[i] += a[k][j] * a[i][j];
		}
		
	}
	cout << endl;
	for (int i = 0; i < n; i++)
	{
		cout.width(5);
		cout << p[i];
	}
}

int search(int **a, int n)
{
	int s = 0;
	int ind = 0;
	for (int i = 0; i < n; i++)
	{         		
		for (int j = 0; j < n; j++)         
			if (*(a[i] + j) < 0)
			{
				s++;
				break;
			}
		if (s == n - 1) 
		{
			ind = i;
			break;
		}
		else s = 0;
	}
	return ind;
}
