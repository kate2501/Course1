// Определить, сколько слов в строке.
#include <iostream>
using namespace std;

const int M = 50;

int main()
{
	char str[M];
	int kol_slov = 1;
	cout << "Vvedi stroku" << endl;
	gets_s(str);
	cout << "isxodnya stroka: " << str << endl;
	puts(str); //вывод так же, как и cout
	int l = strlen(str);
	for (int i = 0; i < l - 1; i++)
		if ((str[i] == ' ') && (str[i + 1] != ' ')) kol_slov++;
	cout << " kol_slov = " << kol_slov << endl;
}
