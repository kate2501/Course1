void Change(char* arr, int n)
{
	for (int i = 0; i < n-1; i++)
	{
		if (arr[i] == '(')
		{
			while (arr[i + 1] != ')')
			{
				arr[i + 1] = '*';
				i++;
			}
			break;
		}
	}
	cout << arr;
}
