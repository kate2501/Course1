int search(char *str,int n )
{
	int max = 0;
	int quant = 0;
	int amount = 0;
	bool flag = false;
	for (int i = 0; i <n; i++)
	{
		if (str[i] != ' ')
		{
			quant++;
			if (quant > max)
			{
				max = quant; 
				flag = true; 
				amount = 0;
			}
		}
		else
		{ 
			if(flag || quant==max) { amount++; flag = false; }
			quant = 0;
		}
	}
	if (quant == max) amount++;
	return amount;
}
